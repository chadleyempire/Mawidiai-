from flask import Flask, render_template, request import os app = Flask(__name__) UPLOAD_FOLDER = "uploads" os.makedirs(UPLOAD_FOLDER, exist_ok=True) coins = 100 @app.route("/") def home(): return render_template("index.html") @app.route("/upload", methods=["GET", "POST"]) def upload(): global coins if request.method == "POST": if coins <= 0: return "Huna coins za upload" file = request.files.get("song") if not file: return "Hakuna file iliyochaguliwa" file.save(os.path.join(UPLOAD_FOLDER, file.filename)) coins -= 10 return f"Upload successful 🎉 Coins zilizosalia: {coins}" return render_template("upload.html") @app.route("/buy-coins") def buy_coins(): global coins coins += 50 return f"Umenunua coins 🎉 Sasa una {coins}" @app.route("/coins") def check_coins(): return f"Wewe una coins: {coins}" if __name__ == "__main__": port = int(os.environ.get("PORT", 10000)) app.run(host="0.0.0.0", port=port)
